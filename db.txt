CREATE TABLE person_form(
id int(10) unsigned NOT NULL AUTO_INCREMENT,
name varchar(255) NOT NULL DEFAULT '',
email varchar(255) NOT NULL DEFAULT '',
birth_year year(4) NOT NULL,
sex tinytext NOT NULL,
number_of_limbs tinyint(1) unsigned NOT NULL,
biography text NOT NULL DEFAULT '',
PRIMARY KEY(id)
);

CREATE TABLE superabillities(
superabillity_name varchar(255) NOT NULL PRIMARY KEY
);

CREATE TABLE person_form_superabillities(
	person_id int(10) unsigned NOT NULL,
	superabillity_name varchar(255) NOT NULL,
	PRIMARY KEY(person_id,superabillity_name)
);

//пример запроса для получения всей информации о записи(включая данные из первой таблицы и таблицы person_form_superabillities)
SELECT p.*, s.*
FROM person_form p
INNER JOIN person_form_superabillities ps
ON ps.person_id = p.id
INNER JOIN superabillities s
ON s.superabillity_name = ps.superabillity_name
ORDER BY p.id
